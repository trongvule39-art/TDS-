-- Khởi tạo UI và dọn dẹp phiên bản cũ...
local CoreGui = game:GetService("CoreGui")
local TweenService = game:GetService("TweenService")

-- Dọn dẹp loader cũ nếu có trùng lặp để tránh chồng chéo giao diện
if CoreGui:FindFirstChild("TrongVuHubLoader") then
CoreGui:FindFirstChild("TrongVuHubLoader"):Destroy()
end

local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "TrongVuHubLoader"
ScreenGui.Parent = CoreGui
ScreenGui.ResetOnSpawn = false

-- Định nghĩa bảng màu đồng bộ Gradient Blue...
local BLUE_SUPER_LIGHT  = Color3.fromRGB(180, 225, 255)
local BLUE_LIGHT        = Color3.fromRGB(0, 160, 255)
local BLUE_MAIN         = Color3.fromRGB(0, 100, 200)
local BLUE_DARK         = Color3.fromRGB(15, 25, 45)
local BLUE_SUPER_DARK  = Color3.fromRGB(8, 12, 25)

-- Tạo khung giao diện chính cho Card thông báo...
local Frame = Instance.new("Frame")
Frame.Parent = ScreenGui
Frame.BackgroundColor3 = BLUE_SUPER_DARK
Frame.BackgroundTransparency = 0.15
Frame.Size = UDim2.new(0, 360, 0, 150)
Frame.Position = UDim2.new(0.5, -180, 0.45, -75) -- Bắt đầu hơi cao một chút để chạy hiệu ứng rơi xuống

local Corner = Instance.new("UICorner")
Corner.CornerRadius = UDim.new(0, 8)
Corner.Parent = Frame

local Stroke = Instance.new("UIStroke")
Stroke.Thickness = 1.5
Stroke.Color = BLUE_MAIN
Stroke.Parent = Frame

-- Tạo nhãn thông báo chính và nút đóng X...
local Label = Instance.new("TextLabel")
Label.Parent = Frame
Label.BackgroundTransparency = 1
Label.Position = UDim2.new(0, 15, 0, 15)
Label.Size = UDim2.new(1, -55, 0, 35)
Label.Font = Enum.Font.SourceSansBold
Label.Text = "Hãy chuyển qua script menu để sử dụng:"
Label.TextColor3 = BLUE_SUPER_LIGHT
Label.TextSize = 15
Label.TextWrapped = true
Label.TextXAlignment = Enum.TextXAlignment.Left

local CloseBtn = Instance.new("TextButton")
CloseBtn.Parent = Frame
CloseBtn.BackgroundTransparency = 1
CloseBtn.Position = UDim2.new(1, -35, 0, 10)
CloseBtn.Size = UDim2.new(0, 25, 0, 25)
CloseBtn.Font = Enum.Font.SourceSansBold
CloseBtn.Text = "X"
CloseBtn.TextColor3 = Color3.fromRGB(255, 100, 100)
CloseBtn.TextSize = 18

CloseBtn.MouseButton1Click:Connect(function()
local fadeOut = TweenService:Create(Frame, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
BackgroundTransparency = 1,
Position = UDim2.new(0.5, -180, 0.55, -75)
})
TweenService:Create(Stroke, TweenInfo.new(0.2), {Transparency = 1}):Play()
fadeOut:Play()
fadeOut.Completed:Connect(function()
ScreenGui:Destroy()
end)
end)

-- Tạo hộp TextBox chứa liên kết loadstring...
local TargetScript = 'loadstring(game:HttpGet("https://raw.githubusercontent.com/trongvule39-art/menu/refs/heads/main/menu.txt", true))()'

local TextBox = Instance.new("TextBox")
TextBox.Parent = Frame
TextBox.BackgroundColor3 = BLUE_DARK
TextBox.Position = UDim2.new(0, 15, 0, 55)
TextBox.Size = UDim2.new(1, -30, 0, 35)
TextBox.Font = Enum.Font.Code
TextBox.Text = TargetScript
TextBox.TextColor3 = Color3.fromRGB(240, 240, 240)
TextBox.TextSize = 10
TextBox.TextWrapped = true
TextBox.TextEditable = false -- Ngăn người dùng sửa nhầm link
TextBox.ClearTextOnFocus = false

local BoxCorner = Instance.new("UICorner")
BoxCorner.CornerRadius = UDim.new(0, 5)
BoxCorner.Parent = TextBox

local BoxStroke = Instance.new("UIStroke")
BoxStroke.Thickness = 1
BoxStroke.Color = Color3.fromRGB(40, 60, 90)
BoxStroke.Parent = TextBox

-- Tạo nút tự động Copy Script siêu nhanh...
local CopyBtn = Instance.new("TextButton")
CopyBtn.Parent = Frame
CopyBtn.BackgroundColor3 = BLUE_MAIN
CopyBtn.Position = UDim2.new(0.04, 0, 0.72, 0)
CopyBtn.Size = UDim2.new(0.92, 0, 0, 30)
CopyBtn.Font = Enum.Font.SourceSansBold
CopyBtn.Text = "COPY SCRIPT"
CopyBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
CopyBtn.TextSize = 14

local CopyCorner = Instance.new("UICorner")
CopyCorner.CornerRadius = UDim.new(0, 5)
CopyCorner.Parent = CopyBtn

CopyBtn.MouseButton1Click:Connect(function()
-- Thực hiện sao chép bằng hàm setclipboard chuẩn của các Executor Roblox
if setclipboard then
setclipboard(TargetScript)
CopyBtn.Text = "ĐÃ COPY VÀO BỘ NHỚ TẠM (CLIPBOARD)!"
CopyBtn.BackgroundColor3 = Color3.fromRGB(46, 117, 89) -- Đổi sang màu xanh lá cây báo thành công
task.wait(1.5)
CopyBtn.Text = "COPY SCRIPT"
CopyBtn.BackgroundColor3 = BLUE_MAIN
else
-- Fallback nếu executor không hỗ trợ setclipboard
CopyBtn.Text = "HÃY TỰ BÔI ĐEN ĐỂ COPY!"
CopyBtn.BackgroundColor3 = Color3.fromRGB(150, 50, 50)
task.wait(1.5)
CopyBtn.Text = "COPY SCRIPT"
CopyBtn.BackgroundColor3 = BLUE_MAIN
end
end)

-- Kích hoạt hiệu ứng xuất hiện mượt mà...
Frame.BackgroundTransparency = 1
Stroke.Transparency = 1
Label.TextTransparency = 1
CloseBtn.TextTransparency = 1
TextBox.BackgroundTransparency = 1
TextBox.TextTransparency = 1
BoxStroke.Transparency = 1
CopyBtn.BackgroundTransparency = 1
CopyBtn.TextTransparency = 1

TweenService:Create(Frame, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
Position = UDim2.new(0.5, -180, 0.5, -75),
BackgroundTransparency = 0.15
}):Play()

TweenService:Create(Stroke, TweenInfo.new(0.5), {Transparency = 0}):Play()
TweenService:Create(Label, TweenInfo.new(0.5), {TextTransparency = 0}):Play()
TweenService:Create(CloseBtn, TweenInfo.new(0.5), {TextTransparency = 0}):Play()
TweenService:Create(TextBox, TweenInfo.new(0.5), {BackgroundTransparency = 0, TextTransparency = 0}):Play()
TweenService:Create(BoxStroke, TweenInfo.new(0.5), {Transparency = 0}):Play()
TweenService:Create(CopyBtn, TweenInfo.new(0.5), {BackgroundTransparency = 0, TextTransparency = 0}):Play()
