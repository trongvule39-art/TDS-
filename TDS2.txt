-- Xử lý sự kiện nút Get Key
GetKeyBtn.MouseButton1Click:Connect(function()
    if setclipboard then
        setclipboard(GET_KEY_LINK)
        GetKeyBtn.Text = "Đã Copy Link!"
        task.wait(1.5)
        GetKeyBtn.Text = "Get Key"
    end
end)

-- Xử lý kiểm tra Key trực tiếp
CheckBtn.MouseButton1Click:Connect(function()
    local success, currentKey = pcall(function()
        return game:HttpGet(KEY_URL):gsub("%s+", "")
    end)
    
    if not success or not currentKey then
        CheckBtn.Text = "LỖI KẾT NỐI SERVER!"
        CheckBtn.BackgroundColor3 = Color3.fromRGB(200, 100, 0)
        task.wait(1.5)
        CheckBtn.Text = "Check Key"
        CheckBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 70)
        return
    end

    if TextBox.Text == currentKey then
        local savedKey = TextBox.Text
        
        -- Cập nhật chữ ĐÚNG KEY ngay lập tức cho người dùng thấy
        CheckBtn.Text = "KEY ĐÚNG! ĐANG TẢI..."
        CheckBtn.BackgroundColor3 = Color3.fromRGB(0, 200, 100)
        
        -- Đợi 1 giây để giao diện kịp hiển thị trạng thái màu xanh
        task.wait(1)
        ScreenGui:Destroy() -- Xóa bảng nhập key đi
        
        task.spawn(function()
            -- Tải script chính tds1 từ GitHub của bạn
            pcall(function()
                loadstring(game:HttpGet("https://raw.githubusercontent.com/trongvule39-art/TowerDefenseSimulator/refs/heads/main/tds1.txt"))()
            end)
        end)

        -- Vòng lặp quét ngầm thời gian thực (Cứ mỗi 10 giây quét một lần)
        task.spawn(function()
            while true do
                task.wait(10)
                local reCheck, latestKey = pcall(function()
                    return game:HttpGet(KEY_URL):gsub("%s+", "")
                end)
                
                if reCheck and latestKey ~= savedKey then
                    game.Players.LocalPlayer:Kick("TrongVu Hub: Key đã bị thay đổi hoặc hết hạn! Vui lòng mở lại script và get key mới.")
                    break
                end
            end
        end)
        
    else
        -- Cập nhật chữ SAI KEY
        CheckBtn.Text = "SAI KEY! THỬ LẠI"
        CheckBtn.BackgroundColor3 = Color3.fromRGB(200, 0, 0)
        task.wait(1.5)
        CheckBtn.Text = "Check Key"
        CheckBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 70)
    end
end)
